64  channel_receive_op(Handler& handler, const IoExecutor& io_ex)
65  : channel_receive<Payload>(&channel_receive_op::do_action),
66  handler_(static_cast<Handler&&>(handler)),
67  work_(handler_, io_ex)

71  static void do_action(channel_operation* base,
72  channel_operation::action a, void* v)
73  {
74  // Take ownership of the operation object.
75  channel_receive_op* o(static_cast<channel_receive_op*>(base));
76  ptr p = { asio::detail::addressof(o->handler_), o, o };
77 
78  ASIO_HANDLER_COMPLETION((*o));
79 
80  // Take ownership of the operation's outstanding work.
81  channel_operation::handler_work<Handler, IoExecutor> w(
82  static_cast<channel_operation::handler_work<Handler, IoExecutor>&&>(
83  o->work_));
84 
85  // Make a copy of the handler so that the memory can be deallocated before
86  // the handler is posted. Even if we're not about to post the handler, a
87  // sub-object of the handler may be the true owner of the memory associated
88  // with the handler. Consequently, a local copy of the handler is required
89  // to ensure that any owning sub-object remains valid until after we have
90  // deallocated the memory here.
91  if (a != channel_operation::destroy_op)
92  {
93  Payload* payload = static_cast<Payload*>(v);
94  asio::detail::completion_payload_handler<Payload, Handler> handler(
95  static_cast<Payload&&>(*payload), o->handler_);
96  p.h = asio::detail::addressof(handler.handler_);
97  p.reset();
98  ASIO_HANDLER_INVOCATION_BEGIN(());
99  if (a == channel_operation::immediate_op)
100  w.immediate(handler, handler.handler_, 0);
101  else if (a == channel_operation::dispatch_op)
102  w.dispatch(handler, handler.handler_);
103  else
104  w.post(handler, handler.handler_);
105  ASIO_HANDLER_INVOCATION_END;
106  }
107  else
108  {
109  asio::detail::binder0<Handler> handler(o->handler_);
110  p.h = asio::detail::addressof(handler.handler_);
111  p.reset();
112  }
113  }
114 
115 private:
116  Handler handler_;
117  channel_operation::handler_work<Handler, IoExecutor> work_;
118 };
