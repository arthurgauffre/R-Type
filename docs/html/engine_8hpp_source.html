36  enum want

38  // Returned by functions to indicate that the engine wants input. The input
39  // buffer should be updated to point to the data. The engine then needs to
40  // be called again to retry the operation.
41  want_input_and_retry = -2,
42 
43  // Returned by functions to indicate that the engine wants to write output.
44  // The output buffer points to the data to be written. The engine then
45  // needs to be called again to retry the operation.

47 
48  // Returned by functions to indicate that the engine doesn't need input or

50  want_nothing = 0,
51

54  // operation is complete, and the engine does not need to be called again.
