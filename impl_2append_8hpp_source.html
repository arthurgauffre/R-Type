32 template <typename Handler, typename... Values>
33 class append_handler
34 {
35 public:
36  typedef void result_type;
37 
38  template <typename H>
39  append_handler(H&& handler, std::tuple<Values...> values)
40  : handler_(static_cast<H&&>(handler)),
41  values_(static_cast<std::tuple<Values...>&&>(values))
42  {
43  }
44 
45  template <typename... Args>
46  void operator()(Args&&... args)
47  {
48  this->invoke(
49  index_sequence_for<Values...>{},
50  static_cast<Args&&>(args)...);
51  }
52 
53  template <std::size_t... I, typename... Args>
54  void invoke(index_sequence<I...>, Args&&... args)
55  {
56  static_cast<Handler&&>(handler_)(
57  static_cast<Args&&>(args)...,
58  static_cast<Values&&>(std::get<I>(values_))...);
59  }

64 };
